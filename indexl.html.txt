        // =========================================================

        window.batchesData = {};

        window.tempImgData = null;
        let financeChart;

        // Vaccine App Logic
        window.vaccineApp = {
            currentDate: new Date(),
            currentCoopId: null,
            scheduleTmpl: [
                { name: 'กัมโบโร', d: 5, m: 'หยอดตา/ปาก' }, { name: 'นิวคาสเซิล/หลอดลม', d: 7, m: 'หยอดตา/จมูก' },

                { name: 'หวัดหน้าบวม (กระตุ้น)', d: 100, m: 'ฉีด' }
            ],
            init() { this.renderCalendar(); },
            generateSchedule(birthDateStr) {
                const b = new Date(birthDateStr);
                return this.scheduleTmpl.map((v, i) => {
                    const d = new Date(b); d.setDate(d.getDate() + v.d);
                    return { id: i, name: v.name, date: d.toISOString().split('T')[0], method: v.m, status: 'pending' };
                });
            },
            renderCoopList() {
                const lst = document.getElementById('vaccineCoopList'); if(!lst) return;
                lst.innerHTML = Object.values(window.batchesData).map(b => `<button onclick="vaccineApp.selectCoop('${b.id}')" class="p-3 rounded-xl border-2 text-sm font-bold text-left transition-colors ${this.currentCoopId===b.id?'border-emerald-500 bg-emerald-50 text-emerald-700':'border-slate-100 bg-white hover:border-emerald-300'}">${b.name}</button>`).join('');
                this.renderCalendar();
            },
            selectCoop(id) { this.currentCoopId = id; this.renderCoopList(); },
            changeMonth(dir) { this.currentDate.setMonth(this.currentDate.getMonth() + dir); this.renderCalendar(); },
            goToToday() { this.currentDate = new Date(); this.renderCalendar(); },
            renderCalendar() {
                const y = this.currentDate.getFullYear(), m = this.currentDate.getMonth();
                const currentMonthLabel = document.getElementById('currentMonthLabel');
                if(currentMonthLabel) currentMonthLabel.textContent = new Date(y, m).toLocaleDateString('th-TH', { month: 'long', year: 'numeric' });
                
                const firstDay = new Date(y, m, 1).getDay();
                const daysInMonth = new Date(y, m + 1, 0).getDate();
                const grid = document.getElementById('calendarGrid'); 
                if(!grid) return;
                grid.innerHTML = '';
                
                for(let i=0; i<firstDay; i++) grid.innerHTML += `<div></div>`;
                
                const b = window.batchesData[this.currentCoopId];
                const today = new Date().toISOString().split('T')[0];

                for(let d=1; d<=daysInMonth; d++) {
                    const dateStr = `${y}-${String(m+1).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
                    let vac = null; if(b && b.vaccinations) vac = b.vaccinations.find(v=>v.date===dateStr);
                    
                    let bg = 'bg-white', tc = 'text-slate-700', border = 'border-slate-100';
                    if(dateStr === today && !vac) { border = 'border-emerald-500'; tc='text-emerald-600 font-bold'; }
                    
                    if(vac) {
                        if(vac.status==='completed') { bg='bg-slate-100'; tc='text-slate-400 line-through'; }
                        else if(dateStr < today) { bg='bg-rose-50'; tc='text-rose-600'; border='border-rose-200'; }
                        else { bg='bg-emerald-50'; tc='text-emerald-700'; border='border-emerald-200'; }

        function startRealtimeSync() {
            if(!user || !db) return;
            onSnapshot(collection(db, 'artifacts', appId, 'users', user.uid, 'batches'), (snap) => {
                snap.docs.forEach(d => window.batchesData[d.id] = d.data());
                renderAll();
                if(document.getElementById('lastSyncTime')) document.getElementById('lastSyncTime').textContent = new Date().toLocaleTimeString('th-TH');
                saveToLocalStorage();
            });
            onSnapshot(collection(db, 'artifacts', appId, 'users', user.uid, 'breeders'), (snap) => {
                window.breedersData = snap.docs.map(d => d.data());
                renderAll();
                saveToLocalStorage();
            });
            onSnapshot(collection(db, 'artifacts', appId, 'users', user.uid, 'finance'), (snap) => {
                window.financeData = snap.docs.map(d => d.data());
                renderAll();
                saveToLocalStorage();
            });
        }

        function saveToLocalStorage() { localStorage.setItem('fp_data', JSON.stringify({ b: window.batchesData, br: window.breedersData, f: window.financeData })); }
        function loadFromLocalStorage() {
            const d = JSON.parse(localStorage.getItem('fp_data') || '{"b":{},"br":[],"f":[]}');
            window.batchesData = d.b; window.breedersData = d.br; window.financeData = d.f;
        }

        window.saveDataToCloud = async (collectionName, docId, data) => {
            if (user && db) {
                try { await setDoc(doc(db, 'artifacts', appId, 'users', user.uid, collectionName, docId), data); } 
                catch(e) { window.showNotification("ข้อผิดพลาดในการบันทึกบน Cloud", "error"); console.error(e); }
            } else {
                if(collectionName==='batches') window.batchesData[docId] = data;
                else if(collectionName==='breeders') { const i = window.breedersData.findIndex(x=>x.id===docId); if(i>-1) window.breedersData[i]=data; else window.breedersData.push(data); }
                else if(collectionName==='finance') window.financeData.push(data);
                saveToLocalStorage(); renderAll(); window.showNotification("บันทึกลงเครื่องแล้ว");
            }
        };

        window.deleteDataFromCloud = async (collectionName, docId) => {
            if (user && db) {
                try { await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, collectionName, docId)); } catch(e) {}
            } else {
                if(collectionName==='batches') delete window.batchesData[docId];
                else if(collectionName==='breeders') window.breedersData = window.breedersData.filter(x=>x.id!==docId);
                else if(collectionName==='finance') window.financeData = window.financeData.filter(x=>x.id!==docId);
                saveToLocalStorage(); renderAll(); window.showNotification("ลบสำเร็จ");
            }
        };

            const file = e.target.files[0]; if(!file) return;

            reader.onload = (ev) => {
                const img = new Image(); img.src = ev.target.result;
                img.onload = () => {
                    const cvs = document.createElement('canvas'); const MAX = 600;
                    let w = img.width, h = img.height;
                    if(w>MAX) { h*=MAX/w; w=MAX; } cvs.width=w; cvs.height=h;
                    cvs.getContext('2d').drawImage(img, 0, 0, w, h);
                    const data = cvs.toDataURL('image/jpeg', 0.6);
                    document.getElementById(previewId).innerHTML = `<img src="${data}" class="w-full h-full object-cover">`;
                    document.getElementById(previewId).classList.remove('hidden');
                    window.tempImgData = data;
                };
            };
        };
